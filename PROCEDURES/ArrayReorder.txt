
Function ArrayReorder(var As Variant, ParamArray Order() As Variant)
    Dim out As Variant: ReDim out(0 To UBound(Order))
    Dim element
    Dim counter:    counter = 0
    For Each element In Order
        out(counter) = var(element - 1)
        counter = counter + 1
    Next
    ArrayReorder = out
End Function
