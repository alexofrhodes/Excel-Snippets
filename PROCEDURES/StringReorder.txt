
Function StringReorder(str As String, delim As String, ParamArray Order() As Variant)
    Dim var:    var = Split(str, delim)
    Dim out As String
    Dim element
    For Each element In Order
        out = out & IIf(out <> "", delim, "") & var(element - 1)
    Next
    StringReorder = out
End Function
