
Public Sub ArraySwapRows(ByRef arr As Variant, ByVal idx1 As Long, idx2 As Long)
     '<<<<Need some Conditional Check ..Like Idx is greater than Ubound or Less than Lbound
    'This sub swap element of an array by row in place.Need 2D array..
    'This is applicable for 2D array.
    Dim el As Variant                      'For temporary data holding..
    Dim CurrentColumn As Long           'This one for running loop

    If IsArray(arr) = False Then                                                'Check if pass argument is an array or not..
        MsgBox "Need array to swap rows"
        Exit Sub
    End If

    If NumberOfArrayDimensions(arr:=arr) <> 2 Then          'Check array dimension.
        MsgBox "Input array must need to be a 2D array"
        Exit Sub
    End If

    If idx1 > UBound(arr, 1) Or idx2 > UBound(arr, 1) Or idx1 < LBound(arr, 1) Or idx2 < LBound(arr, 1) Then  'Check if Index is in range of lower bound and upper bound.
        MsgBox "Check passed index and Lower bound and upper bound"
        Exit Sub
    End If

    If idx1 = idx2 Then Exit Sub            'If both Idx are equal then just exit sub.

    For CurrentColumn = LBound(arr, 2) To UBound(arr, 2)
        el = arr(idx1, CurrentColumn)       'Hold data to the temporary variable.
        arr(idx1, CurrentColumn) = arr(idx2, CurrentColumn)           'Swip value.
        arr(idx2, CurrentColumn) = el                                         'Replace the value..
    Next CurrentColumn
End Sub
