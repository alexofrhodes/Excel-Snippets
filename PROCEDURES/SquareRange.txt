
Sub SquareRange(rng As Range)
    Dim i As Integer
    If rng Is Nothing Then
        If TypeName(Selection) = "Range" Then
            Set rng = Selection
        Else
            Exit Sub
        End If
    End If
    For i = 1 To 4
        With rng
            .Columns.ColumnWidth = _
            .Columns("A").ColumnWidth / .Columns("A").Width * _
            .rows(1).Height
        End With
    Next
End Sub
