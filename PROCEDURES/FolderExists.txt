
Function FolderExists(ByVal strPath As String) As Boolean
'@BlogPosted
'@AssignedModule F_FileFolder
    On Error Resume Next
    FolderExists = ((GetAttr(strPath) And vbDirectory) = vbDirectory)
End Function
