

Function DataFilePartFolder(fileNameWithExtension, Optional IncludeSlash As Boolean) As String
    DataFilePartFolder = left(fileNameWithExtension, InStrRev(fileNameWithExtension, "\") - 1 - IncludeSlash)
End Function


Function DataFilePartFolder(fileNameWithExtension, Optional IncludeSlash As Boolean) As String
    DataFilePartFolder = Left(fileNameWithExtension, InStrRev(fileNameWithExtension, "\") - 1 - IncludeSlash)
End Function
