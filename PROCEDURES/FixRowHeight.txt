Sub FixRowHeight(cell As Range)
'had problem with merged cells so i wrote this
    cell.EntireRow.RowHeight = cell.Parent.StandardHeight * (1 + Len(cell.Cells(1, 1).TEXT) - Len(Replace(cell.Cells(1, 1).TEXT, vbLf, "")))
End Sub
