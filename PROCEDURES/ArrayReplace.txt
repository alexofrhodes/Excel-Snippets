
Function ArrayReplace( _
                     ByVal inputArray As Variant, _
                     this As String, _
                     That As String, _
                     Optional Start As Long = 1, _
                     Optional Count As Long = -1)
    Dim i As Long
    For i = LBound(inputArray) To UBound(inputArray)
        inputArray(i) = Replace(inputArray(i), this, That, Start, Count)
    Next
    ArrayReplace = inputArray
End Function
