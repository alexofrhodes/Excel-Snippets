
Function RangeFindNth(rng As Range, strText As String, occurence As Integer) As Range
    Dim c As Range
    Dim counter As Integer
    For Each c In rng
        If InStr(1, c, strText) > 0 Then counter = counter + 1
        If counter = occurence Then
            Set RangeFindNth = c
            Exit Function
        End If
    Next c
End Function
