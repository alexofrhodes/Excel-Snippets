
Function Sequence(FirstNumberOrDate As Variant, LastNumberOrDate As Variant, _
                Optional DateFormat As String = "yyyy - mm / dd") As Variant
    'pass dates in double quotes eg "1/20/2021"
    Dim i As Long
    Dim Count As Long
    If IsDate(FirstNumberOrDate) And IsDate(LastNumberOrDate) Then
        FirstNumberOrDate = DateAdd("d", -1, FirstNumberOrDate)
        Count = DateDiff("d", FirstNumberOrDate, LastNumberOrDate)
    Else
        Count = LastNumberOrDate
    End If
    Dim arr()
    ReDim Preserve arr(1 To Count)
    For i = 1 To Count
        If IsDate(FirstNumberOrDate) Then
            arr(i) = Format(DateAdd("d", i, CDate(FirstNumberOrDate)), DateFormat)
        Else
            arr(i) = i
        End If
    Next
    Sequence = arr
End Function
