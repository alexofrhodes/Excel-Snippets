
Public Function ProcedureHeaderLineLast( _
                                       Module As VBComponent, _
                                       Procedure As String) As Long
'@BlogPosted
'@AssignedModule F_Vbe_Procedures
'@INCLUDE PROCEDURE ProcedureTitleLineFirst
    If Not ProcedureHeaderExists(Module, Procedure) Then End
    ProcedureHeaderLineLast = ProcedureTitleLineFirst(Module, Procedure) - 1
End Function
