Sub sJoin(ByRef sInput As String, strVar, Optional tail As String = vbNewLine, Optional Head As String = "")
    If TypeName(strVar) = "String" Then
        sInput = IIf(sInput = "", Head & strVar, sInput & tail & Head & strVar)
    Else
        Dim el
        For Each el In strVar
            sJoin sInput, CStr(el), , "head "
        Next
    End If
End Sub
