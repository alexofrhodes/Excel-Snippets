Sub testMacro1()
'@ParentModule A_Test2
'@INCLUDE PROCEDURE TestDoSomeLogging
'@INCLUDE USERFORM zTest
'@INCLUDE DECLARATION MY_NAME
'@INCLUDE DECLARATION myDay
    If Not Day(Now) = myDay.myMonday Then Exit Sub
    TestDoSomeLogging MY_NAME
    Debug.Print formatThis
    zTest.Show
End Sub

Sub TestDoSomeLogging(printThis As String)
'@ParentModule A_Test2
'@INCLUDE PROCEDURE testFormatTextFunction
'@INCLUDE CLASS LogFile
'@INCLUDE DECLARATION testString
    testString = testFormatTextFunction(printThis)
Dim mLogger As New LogFile
mLogger.CreateLogFileByName "testIllustration"
    mLogger.Log testString
    mLogger.OpenLogFile
End Sub

Function testFormatTextFunction(formatThis As String)
'@ParentModule A_Test2
    formatThis = Now & vbTab & formatThis
    testFormatTextFunction = formatThis
End Function
